106     private HandlerExceptionResolver exceptionResolver() {
107         SimpleMappingExceptionResolver exceptionResolver = new SimpleMappingExceptionResolver();
108 
109         Properties exceptionMappings = new Properties();
110 
111         exceptionMappings.put("com.artreset.app.art.exception.ArtNotFoundException", "error/404");
112         exceptionMappings.put("java.lang.Exception", "error/error");
113         exceptionMappings.put("java.lang.RuntimeException", "error/error");
114 
115         exceptionResolver.setExceptionMappings(exceptionMappings);
116 
117         Properties statusCodes = new Properties();
118 
119         statusCodes.put("error/404", "404");
120         statusCodes.put("error/error", "500");
121 
122         exceptionResolver.setStatusCodes(statusCodes);
123 
124         return exceptionResolver;
125     }
126 
127     private MessageSource messageSource() {
128         ResourceBundleMessageSource messageSource = new ResourceBundleMessageSource();
129 
130         messageSource.setBasename(MESSAGE_SOURCE_BASE_NAME);
131         messageSource.setUseCodeAsDefaultMessage(true);
132 
133         return messageSource;
134     }

301     			.description("first description")
302     			.assentialImage(new ImageBuilder().id(1L).build())
303     			.build();
304     	
305     	Art second = new ArtBuilder()
306 		    	.id(1L)
307 		    	.artist(new ArtistBuilder().id(2L).build())

310 		    	.description("second description")
311 		    	.assentialImage(new ImageBuilder().id(2L).build())
312 		    	.build();

316     	mockMvc.perform(get(ArtController.REQUEST_MAPPING_ART_LIST)
317     			)
318     			.andDo(print())
319     			.andExpect(status().isOk())

321     			.andExpect(forwardedUrl("/WEB-INF/views/art/list.jsp"))

333     							hasProperty("title", is("second")),
334     							hasProperty("description", is("second description"))
335     							)
336     					)))
337     			;
