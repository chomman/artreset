1   package com.artreset.common.model;
2   
3   import org.fest.assertions.Assertions;
4   import org.fest.assertions.GenericAssert;
5   import org.joda.time.DateTime;
6   
7   /**
8    * @author Taehyun Jung
9    */
10  public class BaseEntityAssert extends GenericAssert<BaseEntityAssert, BaseEntity> {
11  
12      public BaseEntityAssert(BaseEntity actual) {
13          super(BaseEntityAssert.class, actual);
14      }
15  
16      public static BaseEntityAssert assertThat(BaseEntity actual) {
17          return new BaseEntityAssert(actual);
18      }
19  
20      public BaseEntityAssert creationTimeAndModificationTimeAreEqual() {
21          isNotNull();
22  
23          String errorMessage = String.format(
24                  "Expected creation time to be equal than modification time but were <%s> and <%s>",
25                  actual.getCreationTime(),
26                  actual.getModificationTime()
27          );
28          Assertions.assertThat(actual.getCreationTime())
29                  .overridingErrorMessage(errorMessage)
30                  .isEqualTo(actual.getModificationTime());
31  
32          return this;
33      }
34  
35      public BaseEntityAssert creationTimeIsSet() {
36          isNotNull();
37  
38          String errorMessage = "Expected creation time to be set but was <null>";
39          Assertions.assertThat(actual.getCreationTime()).overridingErrorMessage(errorMessage).isNotNull();
40  
41          return this;
42      }
43  
44      public BaseEntityAssert modificationTimeIsAfterCreationTime() {
45          isNotNull();
46  
47          String errorMessage = String.format(
48                  "Expected modification time to be after creation time but were <%s> and <%s>",
49                  actual.getModificationTime(),
50                  actual.getCreationTime()
51          );
52  
53          DateTime creationTime = actual.getCreationTime();
54          DateTime modificationTime = actual.getModificationTime();
55  
56          Assertions.assertThat(modificationTime.isAfter(creationTime))
57                  .overridingErrorMessage(errorMessage)
58                  .isTrue();
59  
60          return this;
61      }
62  
63      public BaseEntityAssert modificationTimeIsSet() {
64          isNotNull();
65  
66          String errorMessage = "Expected modification time be set was <null>";
67          Assertions.assertThat(actual.getModificationTime())
68                  .overridingErrorMessage(errorMessage)
69                  .isNotNull();
70  
71          return this;
72      }
73  }
