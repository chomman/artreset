23  
24          if (usersConnectionRepository != null) {
25              this.usersConnectionRepositorySet = true;
26          }
27      }
28  
29      @Override
30      public Connection<?> getConnection() {
31          return connection;
32      }
33  
34      @Override
35      void addConnection(String userId) {

37          if (usersConnectionRepositorySet) {
38              super.addConnection(userId);
39          }
40      }
41  
42      public Set<String> getConnections() {
43          return connections;
