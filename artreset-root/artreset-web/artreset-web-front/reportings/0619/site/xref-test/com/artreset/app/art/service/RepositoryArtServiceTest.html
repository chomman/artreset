32  public class RepositoryArtServiceTest {
33  	
34  	private static final String TITLE = "title";
35  
36  	private static final String DESCRIPTION = "description";;
37  
38  	private RepositoryArtService service;
39  	
40  	private ArtRepository repositoryMock;
41  	private ArtistRepository artistRepositoryMock;
42  	
43  	@Before
44  	public void setUp() {
45  		repositoryMock = mock(ArtRepository.class);
46  		service = new RepositoryArtService(repositoryMock, artistRepositoryMock);
47  	}
48  	
49  	@Test
50  	public void add_호출후_새로운_ART_항목이_저장되는지_테스트() {
51  		ArtDTO dto = new ArtDTOBuilder()
52  				.title(TITLE)
53  				.description(DESCRIPTION)
54  				.build();
55  		
56  		service.add(dto);
57  		
58  		ArgumentCaptor<Art> artArgumentCaptor = ArgumentCaptor.forClass(Art.class);
59  		
60  		verify(repositoryMock, times(1)).save(artArgumentCaptor.capture());
61  		verifyNoMoreInteractions(repositoryMock);
62  		
63  		Art model = artArgumentCaptor.getValue();
64  		
65  		assertNull(model.getId());
66  		assertThat(model.getTitle(), is(dto.getTitle()));
67  		assertThat(model.getDescription(), is(dto.getDescription()));
68  	}
69  	
70  	@Test
71  	public void findAll_호출시_모든_Art_항목을_반환하는지_테스트() {
72  		List<Art> models = new ArrayList<Art>();
73  		when(repositoryMock.findAll()).thenReturn(models);
74  		
75  		List<Art> actual = service.findAll();
76  		
77  		verify(repositoryMock, times(1)).findAll();
78  		verifyNoMoreInteractions(repositoryMock);
79  		
80  		assertThat(actual, is(models));
81  	}
